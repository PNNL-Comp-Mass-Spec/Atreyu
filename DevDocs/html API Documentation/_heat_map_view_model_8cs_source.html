353  public double[,] GetCompressedDataInView()
354  {
355  var minScan = this.CurrentMinScan;
356 
357  var exportData = new double[this.dataArray.GetLength(0) + 1, this.dataArray.GetLength(1) + 1];
358 
359  // populate the scan numbers along one axis (the vertical)
360  for (var x = 1; x < exportData.GetLength(0); x++)
361  {
362  var scan = x - 1 + minScan;
363  exportData[x, 0] = scan;
364  }
365 
366  // populate the m/zs on the other axis
367  for (var y = 1; y < exportData.GetLength(1); y++)
368  {
369  var bin = y - 1;
370  var mz = this.BinToMzMap[bin];
371  exportData[0, y] = mz;
372  }
373 
374  // fill the reast of the array with the intensity values (0,0 of the array never assigned, but defaults to "0.0")
375  for (var mz = 1; mz < exportData.GetLength(1); mz++)
376  {
377  for (var scan = 1; scan < exportData.GetLength(0); scan++)
378  {
379  exportData[scan, mz] = this.dataArray[scan - 1, mz - 1];
380  }
381  }
382 
383  return exportData;
384  }
