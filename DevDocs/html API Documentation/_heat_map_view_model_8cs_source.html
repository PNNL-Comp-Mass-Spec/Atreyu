379  public double[,] GetCompressedDataInView()
380  {
381  var minScan = this.CurrentMinScan;
382 
383  var exportData = new double[this.dataArray.GetLength(0) + 1, this.dataArray.GetLength(1) + 1];
384 
385  // populate the scan numbers along one axis (the vertical)
386  for (var x = 1; x < exportData.GetLength(0); x++)
387  {
388  var scan = x - 1 + minScan;
389  exportData[x, 0] = scan;
390  }
391 
392  // populate the m/zs on the other axis
393  for (var y = 1; y < exportData.GetLength(1); y++)
394  {
395  var bin = y - 1;
396  var mz = this.BinToMzMap[bin];
397  exportData[0, y] = mz;
398  }
399 
400  // fill the rest of the array with the intensity values (0,0 of the array never assigned, but defaults to "0.0")
401  for (var mz = 1; mz < exportData.GetLength(1); mz++)
402  {
403  for (var scan = 1; scan < exportData.GetLength(0); scan++)
404  {
405  exportData[scan, mz] = this.dataArray[scan - 1, mz - 1];
406  }
407  }
408 
409  return exportData;
410  }

503  public void UpdateData(double[,] framedata)
504  {
505  if (framedata == null)
506  {
507  return;
508  }
509 
510  var series = this.HeatMapPlotModel.Series[0] as HeatMapSeries;
511  if (series == null)
512  {
513  return;
514  }
515 
516  if (this.ForceMinMaxMz)
517  {
518  this.heatMapPlotModel.Axes[2].AbsoluteMaximum = this.MzWindow.EndBin;
519  this.heatMapPlotModel.Axes[2].AbsoluteMinimum = this.MzWindow.StartBin;
520  }
521  else
522  {
523  this.heatMapPlotModel.Axes[2].AbsoluteMaximum = this.HeatMapData.MaxBins;
524  this.heatMapPlotModel.Axes[2].AbsoluteMinimum = 0;
525  }
526 
527  this.dataArray = framedata;
528 
529  series.Data = this.dataArray;
530 
531  // scans
532  series.X0 = this.CurrentMinScan;
533  series.X1 = this.CurrentMaxScan;
534 
535  // bins
536  series.Y0 = this.CurrentMinBin;
537  series.Y1 = this.CurrentMaxBin;
538 
539  if ((this.CurrentMinScan == 0 && this.CurrentMinBin == 0)
540  || (this.ForceMinMaxMz && this.CurrentMinBin == this.MzWindow.StartBin))
541  {
542  this.heatMapPlotModel.ResetAllAxes();
543  }
544 
545  this.HeatMapPlotModel.InvalidatePlot(true);
546  }
