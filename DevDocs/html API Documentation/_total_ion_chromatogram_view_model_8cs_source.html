297  private void FindPeaks()
298  {
299  this.ticPlotModel.Annotations.Clear();
300  var peakDetector = new PeakDetector();
301 
302  var finderOptions = PeakDetector.GetDefaultSICPeakFinderOptions();
303 
304  List<double> smoothedY;
305 
306  // Create a new dictionary so we don't modify the original one
307  var tempFrameDict = new Dictionary<int, double>(this.uimfData.Scans);
308 
309  for (var i = 0; i < this.uimfData.Scans; i++)
310  {
311  // this is a hack to make the library work and return the proper location index
312  double junk;
313  tempFrameDict.Add(i, this.frameDictionary.TryGetValue(i, out junk) ? junk : 0);
314  }
315 
316  // I am not sure what this does and need to talk to Matt Monroe, but in the example exe file that came with the library
317  // they used half of the length of the list in their previous examples and this seems to work on teh zoomed out version
318  // but not when we zoom in, it seems like an offset problem.
319  var originalpeakLocation = tempFrameDict.Count / 2;
320 
321  // The idea behind this is to always give the key of the mid point of the list, but this causes the finder to blow up.
323  var peaks = peakDetector.FindPeaks(
324  finderOptions,
325  tempFrameDict.OrderBy(x => x.Key).ToList(),
326  originalpeakLocation,
327  out smoothedY);
328 
329  foreach (var peak in peaks)
330  {
332  var index = peak.LocationIndex; // + firstpoint.Key;
333 
334  double intensity;
335  if (!tempFrameDict.TryGetValue(index, out intensity))
336  {
337  intensity = 50;
338  }
339 
340  var halfmax = intensity / 2.0;
341 
342  var peakDataset = tempFrameDict.Where(x => x.Key >= peak.LeftEdge && x.Key <= peak.RightEdge).ToList();
343 
344  // find the left mid point
345  var currPoint = new KeyValuePair<int, double>(0, 0);
346  double leftMidpoint = 0;
347  double rightMidPoint = 0;
348  for (var i = 0; i < peakDataset.Count; i++)
349  {
350  const double Tolerance = 0.01;
351  var prevPoint = currPoint;
352  currPoint = peakDataset[i];
353 
354  if (Math.Abs(leftMidpoint) < Tolerance)
355  {
356  if (smoothedY[currPoint.Key] < halfmax)
357  {
358  continue;
359  }
360 
361  if (Math.Abs(smoothedY[currPoint.Key] - halfmax) < Tolerance)
362  {
363  leftMidpoint = currPoint.Key;
364  continue;
365  }
366 
368  double a1 = prevPoint.Key;
369  double a2 = currPoint.Key;
370  double c = halfmax;
371  double b1 = smoothedY[prevPoint.Key];
372  double b2 = smoothedY[currPoint.Key];
373 
374  leftMidpoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
375  continue;
376  }
377 
378  if (Math.Abs(rightMidPoint) < Tolerance)
379  {
380  if (smoothedY[currPoint.Key] > halfmax)
381  {
382  continue;
383  }
384 
385  if (Math.Abs(smoothedY[currPoint.Key] - halfmax) < Tolerance)
386  {
387  rightMidPoint = currPoint.Key;
388  continue;
389  }
390 
392  double a1 = prevPoint.Key;
393  double a2 = currPoint.Key;
394  double c = halfmax;
395  double b1 = smoothedY[prevPoint.Key];
396  double b2 = smoothedY[currPoint.Key];
397 
398  rightMidPoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
399  }
400  }
401 
402  var resolution = peak.LocationIndex / (rightMidPoint - leftMidpoint);
403 
405  var resolutionString = resolution.ToString("F1", CultureInfo.InvariantCulture);
406 
407  var annotationText = "Peak Location:" + peak.LocationIndex + Environment.NewLine + "Intensity:"
408  + intensity + Environment.NewLine + "Resolution:" + resolutionString;
409 
411  var peakPoint = new OxyPlot.Annotations.PointAnnotation
412  {
413  Text = "R=" + resolutionString,
414  X = peak.LocationIndex,
415  Y = intensity / 2.5,
416  ToolTip = annotationText
417  };
418  this.ticPlotModel.Annotations.Add(peakPoint);
419  }
420  }
