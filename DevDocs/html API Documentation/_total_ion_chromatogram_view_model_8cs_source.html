278  private void FindPeaks()
279  {
280  this.ticPlotModel.Annotations.Clear();
281  var peakDetector = new PeakDetector();
282 
283  var finderOptions = PeakDetector.GetDefaultSICPeakFinderOptions();
284 
285  List<double> smoothedY;
286 
287  // Create a new dictionary so we don't modify the original one
288  var tempFrameDict = new Dictionary<int, double>(this.uimfData.Scans);
289 
290  // this is a hack to make the library work and return the proper location index
291  double junk;
292  for (var i = 0; i < this.uimfData.Scans; i++)
293  {
294  tempFrameDict.Add(i, this.frameDictionary.TryGetValue(i, out junk) ? junk : 0);
295  }
296 
297  // I am not sure what this does and need to talk to Matt Monroe, but in the example exe file that came with the library
298  // they used half of the length of the list in their previous examples and this seems to work on teh zoomed out version
299  // but not when we zoom in, it seems like an offset problem.
300  var originalpeakLocation = tempFrameDict.Count / 2;
301 
302  // The idea behind this is to always give the key of the mid point of the list, but this causes the finder to blow up.
304  var peaks = peakDetector.FindPeaks(
305  finderOptions,
306  tempFrameDict.OrderBy(x => x.Key).ToList(),
307  originalpeakLocation,
308  out smoothedY);
309 
310  foreach (var peak in peaks)
311  {
313  var index = peak.LocationIndex; // + firstpoint.Key;
314 
315  double intensity;
316  if (!tempFrameDict.TryGetValue(index, out intensity))
317  {
318  intensity = 50;
319  }
320 
321  var halfmax = intensity / 2.0;
322 
323  var peakDataset = tempFrameDict.Where(x => x.Key >= peak.LeftEdge && x.Key <= peak.RightEdge).ToList();
324 
325  // find the left mid point
326  var currPoint = new KeyValuePair<int, double>(0, 0);
327  double leftMidpoint = 0;
328  double rightMidPoint = 0;
329  for (var i = 0; i < peakDataset.Count; i++)
330  {
331  const double Tolerance = 0.01;
332  var prevPoint = currPoint;
333  currPoint = peakDataset[i];
334 
335  if (Math.Abs(leftMidpoint) < Tolerance)
336  {
337  if (smoothedY[currPoint.Key] < halfmax)
338  {
339  continue;
340  }
341 
342  if (Math.Abs(smoothedY[currPoint.Key] - halfmax) < Tolerance)
343  {
344  leftMidpoint = currPoint.Key;
345  continue;
346  }
347 
349  double a1 = prevPoint.Key;
350  double a2 = currPoint.Key;
351  double c = halfmax;
352  double b1 = smoothedY[prevPoint.Key];
353  double b2 = smoothedY[currPoint.Key];
354 
355  leftMidpoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
356  continue;
357  }
358 
359  if (Math.Abs(rightMidPoint) < Tolerance)
360  {
361  if (smoothedY[currPoint.Key] > halfmax)
362  {
363  continue;
364  }
365 
366  if (Math.Abs(smoothedY[currPoint.Key] - halfmax) < Tolerance)
367  {
368  rightMidPoint = currPoint.Key;
369  continue;
370  }
371 
373  double a1 = prevPoint.Key;
374  double a2 = currPoint.Key;
375  double c = halfmax;
376  double b1 = smoothedY[prevPoint.Key];
377  double b2 = smoothedY[currPoint.Key];
378 
379  rightMidPoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
380  continue;
381  }
382  }
383 
384  var resolution = peak.LocationIndex / (rightMidPoint - leftMidpoint);
385 
386  var pointAnnotation1 = new OxyPlot.Annotations.PointAnnotation
387  {
388  X = leftMidpoint,
389  Y = halfmax,
390  Text = "Left",
391  ToolTip =
392  "Left mid Point Found at "
393  + leftMidpoint
394  };
395 
397  var pointAnnotation2 = new OxyPlot.Annotations.PointAnnotation
398  {

401  Text = "right",
402  ToolTip =

405  };
