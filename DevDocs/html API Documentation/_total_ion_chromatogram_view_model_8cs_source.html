184  public void UpdateFrameData(double[,] data)
185  {
186  if (data == null)
187  {
188  return;
189  }
190 
191  this.frameData = data;
192 
193  if (this.endScan == 0)
194  {
195  this.startScan = 0;
196  this.endScan = 359;
197  }
198 
199  this.frameDictionary = new Dictionary<int, double>();
200 
201  for (var i = 0; i < this.frameData.GetLength(0); i++)
202  {
203  var index = i + this.startScan;
204  for (var j = 0; j < this.frameData.GetLength(1); j++)
205  {
206  if (this.frameDictionary.ContainsKey(index))
207  {
208  this.frameDictionary[index] += this.frameData[i, j];
209  }
210  else
211  {
212  this.frameDictionary.Add(index, this.frameData[i, j]);
213  }
214  }
215  }
216 
217  var series = this.TicPlotModel.Series[0] as LineSeries;
218 
219  if (series == null)
220  {
221  return;
222  }
223 
224  series.MarkerType = MarkerType.Circle;
225  series.MarkerSize = 2.5;
226 
227  // series.MarkerStrokeThickness = 2;
228  series.MarkerFill = OxyColors.Black;
229  series.BrokenLineColor = OxyColors.Automatic;
230  series.BrokenLineStyle = LineStyle.Dot;
231  series.BrokenLineThickness = 1;
232  series.Points.Clear();
233  foreach (var d in this.frameDictionary)
234  {
235  series.Points.Add(new DataPoint(d.Key, d.Value));
236  series.Points.Add(new DataPoint(double.NaN, double.NaN));
237  }
238 
239  this.FindPeaks();
240 
241  this.TicPlotModel.InvalidatePlot(true);
242  }

294  private void FindPeaks()
295  {
296  this.ticPlotModel.Annotations.Clear();
297 
298  // Create a new dictionary so we don't modify the original one
299  var tempFrameDict = new Dictionary<double, double>(this.uimfData.Scans);
300 
301  for (var i = 0; i < this.uimfData.Scans; i++)
302  {
303  // this is a hack to make the library work and return the proper location index
304  double junk;
305  tempFrameDict.Add(i, this.frameDictionary.TryGetValue(i, out junk) ? junk : 0);
306  }
307 
308  var results = Utilities.PeakFinder.FindPeaks(tempFrameDict.ToList());
309 
310  foreach (var peakInformation in results.Peaks)
311  {
312  var resolutionString = peakInformation.ResolvingPower.ToString("F1", CultureInfo.InvariantCulture);
313 
314  var peakPoint = new OxyPlot.Annotations.PointAnnotation
315  {
316  Text = "R=" + resolutionString,
317  X = peakInformation.PeakCenter,
318  Y = peakInformation.Intensity / 2.5,
319  ToolTip = peakInformation.ToString()
320  };
321  this.ticPlotModel.Annotations.Add(peakPoint);
322  }
323  }
