114  public static FileInfo GetOutputLocation(
115  FileInfo originFile,
116  string dataType,
117  int frameNumber,
118  string fileExtension = "csv")
119  {
120  var locationRelativeToInput = originFile.FullName.Substring(inputDirectory.FullName.Length + 1);
121  var nestedLocation = Path.Combine(outputDirectory.FullName, locationRelativeToInput);
122  var csvFullPath = Path.ChangeExtension(nestedLocation, fileExtension);
123  var oldName = new FileInfo(csvFullPath);
124 
125  var oldDir = oldName.DirectoryName;
126  var newName = Path.GetFileNameWithoutExtension(csvFullPath);
127 
128  newName += "_" + dataType;
129 
130  if (frameNumber >= 0)
131  {
132  newName += "_" + frameNumber.ToString("0000");
133  }
134 
135  newName += Path.GetExtension(oldName.FullName);
136 
137  return oldDir == null ? null : new FileInfo(Path.Combine(oldDir, newName));
138  }

266  public static void OutputHeatMap(double[,] data, FileInfo outputFile, bool verbose)
267  {
268  using (var stream = GetFileStream(outputFile))
269  {
270  if (stream == null)
271  {
272  PrintFileCreationError(outputFile.FullName);
273  return;
274  }
275 
276  for (var bin = 0; bin < data.GetLength(1); bin++)
277  {
278  var content = string.Empty;
279  for (var scan = 0; scan < data.GetLength(0); scan++)
280  {
281  content += data[scan, bin].ToString("0.00") + ",";
282  }
283 
284  stream.WriteLine(content);
285  }
286 
287  if (verbose)
288  {
289  Console.WriteLine("flushing data to file " + outputFile.FullName);
290  }
291  }
292  }

437  private static StreamWriter GetFileStream(FileInfo outputFile)
438  {
439  var outstring = outputFile.DirectoryName;
440  if (outstring == null)
441  {
442  Console.WriteLine();
443  Console.Error.WriteLine(
444  "ERROR: We will expand upong this later, but we couldn't find the directory of the output file and it will not be output");
445  Console.WriteLine();
446  return null;
447  }
448 
449  var outDirectory = new DirectoryInfo(outstring);
450 
451  if (!outDirectory.Exists)
452  {
453  outDirectory.Create();
454  }
455 
456  if (outputFile.Exists)
457  {
458  Console.WriteLine();
459  Console.WriteLine(
460  "One of the output files (" + outputFile.FullName
461  + ") already exists, so we are deleting it Mwahahaha!");
462  Console.WriteLine();
463  outputFile.Delete();
464  }
465 
466  return outputFile.CreateText();
467  }
