158  Heading =
159  new HeadingInfo(
160  "UIMF Data Extractor",
161  typeof(Program).Assembly.GetName().Version.ToString()),
162  Copyright = new CopyrightInfo("PNNL", 2015),
163  AdditionalNewLineAfterOption = true,
164  AddDashesToOption = true
165  };
166 
167  help.AddPreOptionsLine(string.Empty);
168  help.AddPreOptionsLine(" This application batch processes Unified Ion Mobility Files (UIMF)");
169  help.AddPreOptionsLine(" to output the raw data into Comma Seperated Value (csv) format");
170  help.AddPreOptionsLine(string.Empty);
171  help.AddPreOptionsLine(" Usage:");
172  help.AddPreOptionsLine(" UIMFDataExtractor.exe -i SOURCEFOLDER is the minimum requirement to run");
173  help.AddPreOptionsLine(" If you do not specifiy an output format (m/z or tic) than the ");
174  help.AddPreOptionsLine(" program will simply print what files it found.");
175  help.AddPreOptionsLine(" If no output directory is specified, then it will default to the same");
176  help.AddPreOptionsLine(" folder as the UIMF");
177 
178  if (this.LastParserState.Errors.Any())
179  {
180  var errors = help.RenderParsingErrorsText(this, 2); // indent with two spaces
181 
182  if (!string.IsNullOrEmpty(errors))
183  {
184  help.AddPreOptionsLine(string.Concat(Environment.NewLine, "ERROR(S):"));
185  help.AddPreOptionsLine(errors);
186  }
187  }
188 
189  help.AddOptions(this);
190 
191  return help;
192  }
