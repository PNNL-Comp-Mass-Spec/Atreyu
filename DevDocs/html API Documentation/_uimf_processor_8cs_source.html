153  private static PeakSet FindPeaks(IReadOnlyList<KeyValuePair<double, double>> dataList)
154  {
155  var datapointList = new PeakSet();
156  const int Precision = 100000;
157 
158  var peakDetector = new PeakDetector();
159 
160  var finderOptions = PeakDetector.GetDefaultSICPeakFinderOptions();
161 
162  List<double> smoothedY;
163 
164  // Create a new dictionary so we don't modify the original one
165  var tempFrameList = new List<KeyValuePair<int, double>>(dataList.Count);
166 
167  // We have to give it integers, but we need the mz, so we will multiply the mz by the precision
168  // and later get the correct value back by dividing it out again
169  for (var i = 0; i < dataList.Count; i++)
170  {
171  tempFrameList.Add(new KeyValuePair<int, double>((int)(dataList[i].Key * Precision), dataList[i].Value));
172  }
173 
174  // I am not sure what this does but in the example exe file that came with the library
175  // they used half of the length of the list in their previous examples and this seems to work
176  var originalpeakLocation = tempFrameList.Count / 2;
177 
178  var allPeaks = peakDetector.FindPeaks(
179  finderOptions,
180  tempFrameList.OrderBy(x => x.Key).ToList(),
181  originalpeakLocation,
182  out smoothedY);
183 
185  foreach (var peak in allPeaks)
186  {
187  const double Tolerance = 0.01;
188  var centerPoint = tempFrameList.ElementAt(peak.LocationIndex);
189  var offsetCenter = centerPoint.Key; // + firstpoint.Key;
190  var intensity = centerPoint.Value;
191  var smoothedPeakIntensity = smoothedY[peak.LocationIndex];
192 
193  var realCenter = (double)offsetCenter / Precision;
194  var halfmax = smoothedPeakIntensity / 2.0;
195 
196  var currPoint = new KeyValuePair<int, double>(0, 0);
197  var currPointIndex = 0;
198  double leftMidpoint = 0;
199  double rightMidPoint = 0;
200 
201  var allPoints = new List<PointInformation>();
202  var leftSidePeaks = new List<KeyValuePair<int, double>>();
203  for (var l = peak.LeftEdge; l < peak.LocationIndex && l < tempFrameList.Count; l++)
204  {
205  leftSidePeaks.Add(tempFrameList[l]);
206  allPoints.Add(
207  new PointInformation
208  {
209  Location = (double)tempFrameList[l].Key / Precision,
210  Intensity = tempFrameList[l].Value,
211  SmoothedIntensity = smoothedY[l]
212  });
213  }
214 
215  var rightSidePeaks = new List<KeyValuePair<int, double>>();
216  for (var r = peak.LocationIndex; r < peak.RightEdge && r < tempFrameList.Count; r++)
217  {
218  rightSidePeaks.Add(tempFrameList[r]);
219  allPoints.Add(
220  new PointInformation
221  {
222  Location = (double)tempFrameList[r].Key / Precision,
223  Intensity = tempFrameList[r].Value,
224  SmoothedIntensity = smoothedY[r]
225  });
226  }
227 
228  // find the left side half max
229  foreach (var leftSidePeak in leftSidePeaks)
230  {
231  var prevPoint = currPoint;
232  currPoint = leftSidePeak;
233  var prevPointIndex = currPointIndex;
234 
235  currPointIndex = tempFrameList.BinarySearch(
236  currPoint,
237  Comparer<KeyValuePair<int, double>>.Create((left, right) => left.Key - right.Key));
238 
239  if (smoothedY[currPointIndex] < halfmax)
240  {
241  continue;
242  }
243 
244  if (Math.Abs(smoothedY[currPointIndex] - halfmax) < Tolerance)
245  {
246  leftMidpoint = currPoint.Key;
247  continue;
248  }
249 
251  double a1 = prevPoint.Key;
252  double a2 = currPoint.Key;
253  double c = halfmax;
254  double b1 = smoothedY[prevPointIndex];
255  double b2 = smoothedY[currPointIndex];
256 
257  leftMidpoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
258  break;
259  }
260 
261  // find the right side of the half max
262  foreach (var rightSidePeak in rightSidePeaks)
263  {
264  var prevPoint = currPoint;
265  currPoint = rightSidePeak;
266  var prevPointIndex = currPointIndex;
267  currPointIndex = tempFrameList.BinarySearch(
268  currPoint,
269  Comparer<KeyValuePair<int, double>>.Create((left, right) => left.Key - right.Key));
270 
271  if (smoothedY[currPointIndex] > halfmax || smoothedY[currPointIndex] < 0)
272  {
273  continue;
274  }
275 
276  if (Math.Abs(smoothedY[currPointIndex] - halfmax) < Tolerance)
277  {
278  rightMidPoint = currPoint.Key;
279  continue;
280  }
281 
283  double a1 = prevPoint.Key;
284  double a2 = currPoint.Key;
285  double c = halfmax;
286  double b1 = smoothedY[prevPointIndex];
287  double b2 = smoothedY[currPointIndex];
288 
289  rightMidPoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
290  break;
291  }
292 
293  var correctedRightMidPoint = rightMidPoint / Precision;
294  var correctedLeftMidPoint = leftMidpoint / Precision;
295  var fullWidthHalfMax = correctedRightMidPoint - correctedLeftMidPoint;
296  var resolution = realCenter / fullWidthHalfMax;
297 
298  var temp = new PeakInformation
299  {
300  AreaUnderThePeak = peak.Area,
301  FullWidthHalfMax = fullWidthHalfMax,
302  Intensity = intensity,
303  LeftMidpoint = correctedLeftMidPoint,
304  PeakCenter = realCenter,
305  RightMidpoint = correctedRightMidPoint,
306  ResolvingPower = resolution,
307  SmoothedIntensity = smoothedPeakIntensity,
308  TotalDataPointSet = allPoints
309  };
310 
311  if (temp.ResolvingPower > 0 && !double.IsInfinity(temp.ResolvingPower))
312  {
313  datapointList.Peaks.Add(temp);
314  }
315  }
316 
317  return datapointList;
318

397  private static List<KeyValuePair<double, int>> GetFullMzInfo(DataReader uimf, int frameNumber)
398  {
399  var frameParams = uimf.GetFrameParams(frameNumber);
400  var maxScans = uimf.GetFrameParams(frameNumber).Scans;
401 
402  var typeString = frameParams.GetValue(FrameParamKeyType.FrameType);
403 
404  if (string.IsNullOrWhiteSpace(typeString))

653  private static List<KeyValuePair<double, double>> GetXicInfo(
654  DataReader uimf,
655  int frameNumber,
656  double xicMz,
657  double tolerance,
658  bool getMsms)
659  {
660  const DataReader.ToleranceType Tolerance = DataReader.ToleranceType.Thomson;
661 
662  var frametype = getMsms ? DataReader.FrameType.MS2 : DataReader.FrameType.MS1;
663 
664  if (!uimf.DoesContainBinCentricData())
665  {
666  Console.WriteLine(uimf.UimfFilePath + " Does not have bin centric data which is required to get XiC");
667  Console.WriteLine("starting to create it, this may take some time");
668  var fileName = uimf.UimfFilePath;
669  uimf.Dispose();
670 
671  var dataWriter = new DataWriter(fileName);
672  dataWriter.CreateBinCentricTables();
673  dataWriter.Dispose();
674 
675  uimf = new DataReader(fileName);
676  Console.WriteLine("Finished Creating bin centric tables for " + uimf.UimfFilePath);
677  }
678 
679  List<IntensityPoint> xic;
680  try
681  {
682  xic = uimf.GetXic(xicMz, tolerance, frametype, Tolerance);
683  }
684  catch (Exception)
685  {
686  Console.Error.WriteLine("Unable to get XiC on first attempt for " + uimf.UimfFilePath);
687  var tempreader = new DataReader(uimf.UimfFilePath);
688 
689  try
690  {
691  xic = tempreader.GetXic(xicMz, tolerance, frametype, Tolerance);
692  }
693  catch (Exception)
694  {
695  Console.Error.WriteLine(
696  "Unable to get XiC on second attempt for " + uimf.UimfFilePath + ", we are not trying again.");
697  return null;
698  }
699  }
700 
701  var frameData = xic.Where(point => point.ScanLc == frameNumber - 1);
702 
703  var data = new List<KeyValuePair<double, double>>();
704 
705  foreach (var intensityPoint in frameData)
706  {
707  var driftTime = uimf.GetDriftTime(intensityPoint.ScanLc + 1, intensityPoint.ScanIms, true);
708  data.Add(new KeyValuePair<double, double>(driftTime, intensityPoint.Intensity));
709  }
710 
711  return data;
712  }

907  ProcessFrame(uimfReader, file, i);

910  else

912  ProcessFrame(uimfReader, file);
913  }
914  }
