202  private static PeakSet FindPeaks(IReadOnlyList<KeyValuePair<double, double>> dataList)
203  {
204  var datapointList = new PeakSet();
205  const int Precision = 100000;
206 
207  var peakDetector = new PeakDetector();
208 
209  var finderOptions = PeakDetector.GetDefaultSICPeakFinderOptions();
210 
211  List<double> smoothedY;
212 
213  // Create a new dictionary so we don't modify the original one
214  var tempFrameList = new List<KeyValuePair<int, double>>(dataList.Count);
215 
216  // We have to give it integers, but we need the mz, so we will multiply the mz by the precision
217  // and later get the correct value back by dividing it out again
218  for (var i = 0; i < dataList.Count; i++)
219  {
220  tempFrameList.Add(new KeyValuePair<int, double>((int)(dataList[i].Key * Precision), dataList[i].Value));
221  }
222 
223  // I am not sure what this does but in the example exe file that came with the library
224  // they used half of the length of the list in their previous examples and this seems to work
225  var originalpeakLocation = tempFrameList.Count / 2;
226 
227  var allPeaks = peakDetector.FindPeaks(
228  finderOptions,
229  tempFrameList.OrderBy(x => x.Key).ToList(),
230  originalpeakLocation,
231  out smoothedY);
232 
234  foreach (var peak in allPeaks)
235  {
236  const double Tolerance = 0.01;
237  var centerPoint = tempFrameList.ElementAt(peak.LocationIndex);
238  var offsetCenter = centerPoint.Key; // + firstpoint.Key;
239  var intensity = centerPoint.Value;
240  var smoothedPeakIntensity = smoothedY[peak.LocationIndex];
241 
242  var realCenter = (double)offsetCenter / Precision;
243  var halfmax = smoothedPeakIntensity / 2.0;
244 
245  var currPoint = new KeyValuePair<int, double>(0, 0);
246  var currPointIndex = 0;
247  double leftMidpoint = 0;
248  double rightMidPoint = 0;
249 
250  var allPoints = new List<PointInformation>();
251  var leftSidePeaks = new List<KeyValuePair<int, double>>();
252  for (var l = peak.LeftEdge; l < peak.LocationIndex && l < tempFrameList.Count; l++)
253  {
254  leftSidePeaks.Add(tempFrameList[l]);
255  allPoints.Add(
256  new PointInformation
257  {
258  Location = (double)tempFrameList[l].Key / Precision,
259  Intensity = tempFrameList[l].Value,
260  SmoothedIntensity = smoothedY[l]
261  });
262  }
263 
264  var rightSidePeaks = new List<KeyValuePair<int, double>>();
265  for (var r = peak.LocationIndex; r < peak.RightEdge && r < tempFrameList.Count; r++)
266  {
267  rightSidePeaks.Add(tempFrameList[r]);
268  allPoints.Add(
269  new PointInformation
270  {
271  Location = (double)tempFrameList[r].Key / Precision,
272  Intensity = tempFrameList[r].Value,
273  SmoothedIntensity = smoothedY[r]
274  });
275  }
276 
277  // find the left side half max
278  foreach (var leftSidePeak in leftSidePeaks)
279  {
280  var prevPoint = currPoint;
281  currPoint = leftSidePeak;
282  var prevPointIndex = currPointIndex;
283 
284  currPointIndex = tempFrameList.BinarySearch(
285  currPoint,
286  Comparer<KeyValuePair<int, double>>.Create((left, right) => left.Key - right.Key));
287 
288  if (smoothedY[currPointIndex] < halfmax)
289  {
290  continue;
291  }
292 
293  if (Math.Abs(smoothedY[currPointIndex] - halfmax) < Tolerance)
294  {
295  leftMidpoint = currPoint.Key;
296  continue;
297  }
298 
300  double a1 = prevPoint.Key;
301  double a2 = currPoint.Key;
302  double c = halfmax;
303  double b1 = smoothedY[prevPointIndex];
304  double b2 = smoothedY[currPointIndex];
305 
306  leftMidpoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
307  break;
308  }
309 
310  // find the right side of the half max
311  foreach (var rightSidePeak in rightSidePeaks)
312  {
313  var prevPoint = currPoint;
314  currPoint = rightSidePeak;
315  var prevPointIndex = currPointIndex;
316  currPointIndex = tempFrameList.BinarySearch(
317  currPoint,
318  Comparer<KeyValuePair<int, double>>.Create((left, right) => left.Key - right.Key));
319 
320  if (smoothedY[currPointIndex] > halfmax || smoothedY[currPointIndex] < 0)
321  {
322  continue;
323  }
324 
325  if (Math.Abs(smoothedY[currPointIndex] - halfmax) < Tolerance)
326  {
327  rightMidPoint = currPoint.Key;
328  continue;
329  }
330 
332  double a1 = prevPoint.Key;
333  double a2 = currPoint.Key;
334  double c = halfmax;
335  double b1 = smoothedY[prevPointIndex];
336  double b2 = smoothedY[currPointIndex];
337 
338  rightMidPoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
339  break;
340  }
341 
342  var correctedRightMidPoint = rightMidPoint / Precision;
343  var correctedLeftMidPoint = leftMidpoint / Precision;
344  var fullWidthHalfMax = correctedRightMidPoint - correctedLeftMidPoint;
345  var resolution = realCenter / fullWidthHalfMax;
346 
347  var temp = new PeakInformation
348  {
349  AreaUnderThePeak = peak.Area,
350  FullWidthHalfMax = fullWidthHalfMax,
351  Intensity = intensity,
352  LeftMidpoint = correctedLeftMidPoint,
353  PeakCenter = realCenter,
354  RightMidpoint = correctedRightMidPoint,
355  ResolvingPower = resolution,
356  SmoothedIntensity = smoothedPeakIntensity,
357  TotalDataPointSet = allPoints
358  };
359 
360  if (temp.ResolvingPower > 0 && !double.IsInfinity(temp.ResolvingPower))
361  {
362  datapointList.Peaks.Add(temp);
363  }
364  }
365 
366  return datapointList;
367

702  private static List<KeyValuePair<double, double>> GetXicInfo(
703  DataReader uimf,
704  int frameNumber,
705  double xicMz,
706  double tolerance,
707  bool getMsms)
708  {
709  const DataReader.ToleranceType Tolerance = DataReader.ToleranceType.Thomson;
710 
711  var frametype = getMsms ? DataReader.FrameType.MS2 : DataReader.FrameType.MS1;
712 
713  if (!uimf.DoesContainBinCentricData())
714  {
715  Console.WriteLine(uimf.UimfFilePath + " Does not have bin centric data which is required to get XiC");
716  Console.WriteLine("starting to create it, this may take some time");
717  var fileName = uimf.UimfFilePath;
718  uimf.Dispose();
719 
720  var dataWriter = new DataWriter(fileName);
721  dataWriter.CreateBinCentricTables();
722  dataWriter.Dispose();
723 
724  uimf = new DataReader(fileName);
725  Console.WriteLine("Finished Creating bin centric tables for " + uimf.UimfFilePath);
726  }
727 
728  List<IntensityPoint> xic;
729  try
730  {
731  xic = uimf.GetXic(xicMz, tolerance, frametype, Tolerance);
732  }
733  catch (Exception)
734  {
735  Console.Error.WriteLine("Unable to get XiC on first attempt for " + uimf.UimfFilePath);
736  var tempreader = new DataReader(uimf.UimfFilePath);
737 
738  try
739  {
740  xic = tempreader.GetXic(xicMz, tolerance, frametype, Tolerance);
741  }
742  catch (Exception)
743  {
744  Console.Error.WriteLine(
745  "Unable to get XiC on second attempt for " + uimf.UimfFilePath + ", we are not trying again.");
746  return null;
747  }
748  }
749 
750  var frameData = xic.Where(point => point.ScanLc == frameNumber - 1);
751 
752  var data = new List<KeyValuePair<double, double>>();
753 
754  foreach (var intensityPoint in frameData)
755  {

758  }
