158  private static DataReader.FrameType GetFrameType(string frameType)
159  {
160  var temp = frameType.ToLower();
161  switch (temp)

211  private static List<KeyValuePair<double, int>> GetFullMzInfo(DataReader uimf, int frameNumber)
212  {
213  var frameParams = uimf.GetFrameParams(frameNumber);
214  var maxScans = uimf.GetFrameParams(frameNumber).Scans;
215 
216  var typeString = frameParams.GetValue(FrameParamKeyType.FrameType);
217 
218  if (string.IsNullOrWhiteSpace(typeString))
219  {
220  Console.Error.WriteLine(
221  "ERROR: Had a problem getting the frame type which means we can't get the MZ data");
222  return null;
223  }
224 
225  DataReader.FrameType frameType;
226  var type = frameParams.GetValue(FrameParamKeyType.FrameType);
227 
228  try
229  {
230  frameType = GetFrameType(type);
231  }
232  catch (NotImplementedException)
233  {
234  Console.Error.WriteLine(
235  "ERROR: An Unknown Frame Type was read on Frame: {0}, we wil continue as best we can by defaulting to MS1. The Type read was {1}.",
236  frameNumber,
237  type);
238  frameType = DataReader.FrameType.MS1;
239  }
240 
241  double[] mzs;
242  int[] intensities;
243  uimf.GetSpectrum(frameNumber, frameNumber, frameType, 1, maxScans, out mzs, out intensities);
244  var data = new List<KeyValuePair<double, int>>(mzs.Length);
245  for (var i = 0; i < mzs.Length && i < intensities.Length; i++)
246  {
247  data.Add(new KeyValuePair<double, int>(mzs[i], intensities[i]));
248  }
249 
250  return data;
251  }

494  private static List<KeyValuePair<double, double>> GetXicInfo(
495  DataReader uimf,
496  int frameNumber,
497  double xicMz,
498  double tolerance,
499  bool getMsms)
500  {
501  const DataReader.ToleranceType Tolerance = DataReader.ToleranceType.Thomson;
502 
503  var frametype = getMsms ? DataReader.FrameType.MS2 : DataReader.FrameType.MS1;
504 
505  if (!uimf.DoesContainBinCentricData())
506  {
507  Console.WriteLine(uimf.UimfFilePath + " Does not have bin centric data which is required to get XiC");
508  Console.WriteLine("starting to create it, this may take some time");
509  var fileName = uimf.UimfFilePath;
510  uimf.Dispose();
511 
512  var dataWriter = new DataWriter(fileName);
513  dataWriter.CreateBinCentricTables();
514  dataWriter.Dispose();
515 
516  uimf = new DataReader(fileName);
517  Console.WriteLine("Finished Creating bin centric tables for " + uimf.UimfFilePath);
518  }
519 
520  List<IntensityPoint> xic;
521  try
522  {
523  xic = uimf.GetXic(xicMz, tolerance, frametype, Tolerance);
524  }
525  catch (Exception)
526  {
527  Console.Error.WriteLine("Unable to get XiC on first attempt for " + uimf.UimfFilePath);
528  var tempreader = new DataReader(uimf.UimfFilePath);
529 
530  try
531  {
532  xic = tempreader.GetXic(xicMz, tolerance, frametype, Tolerance);
533  }
534  catch (Exception)
535  {
536  Console.Error.WriteLine(
537  "Unable to get XiC on second attempt for " + uimf.UimfFilePath + ", we are not trying again.");
538  return null;
539  }
540  }
541 
542  var frameData = xic.Where(point => point.ScanLc == frameNumber - 1);
543 
544  var data = new List<KeyValuePair<double, double>>();
545 
546  // I think this is more readable with a regular loop than a very long linq query
547  // ReSharper disable once LoopCanBeConvertedToQuery
548  foreach (var intensityPoint in frameData)
549  {
550  var driftTime = uimf.GetDriftTime(intensityPoint.ScanLc + 1, intensityPoint.ScanIms, true);
551  data.Add(new KeyValuePair<double, double>(driftTime, intensityPoint.Intensity));
552  }
553 
554  return data;
555  }

677  private static void ProcessFrame(DataReader uimf, FileInfo originFile, int frameNumber = 1)
678  {
679  // eventually I will also have methods for mz and heatmap,
681  var frameParams = uimf.GetFrameParams(frameNumber);
682 
683  if (frameParams == null)
684  {
685  // Frame number is out of range
686  Console.WriteLine();

688  "ERROR: Somehow a frame number that doesn't exist was attempted to read" + Environment.NewLine
689  + "So we are not creating data for frame " + frameNumber + " Of " + Environment.NewLine

691  Console.WriteLine();
692  return;

694 
695  if (options.GetHeatmap)

704  GetMz(uimf, originFile, frameNumber);

706 
707  if (options.GetTiC)

710  }
711 
712  if (options.GetXiC > 0)
713  {
714  GetXic(uimf, originFile, frameNumber);
715  }
716 
717  if (options.Verbose)
718  {
719  Console.WriteLine("Finished processing Frame " + frameNumber + " of " + originFile.FullName);

729  private static void ProcessUimf(FileInfo file)
730  {
731  if (!file.Exists)
732  {
733  PrintNotFoundError(file);
734  return;
735  }
736 
737  using (var uimfReader = new DataReader(file.FullName))
738  {
739  if (options.Verbose)
740  {
741  Console.WriteLine("Starting to process " + file.FullName);
742  }
743 
744  var framecount = uimfReader.GetGlobalParams().NumFrames;
745  if (options.AllFrames)
746  {
747  // ReSharper disable once AccessToDisposedClosure
748  for (var i = 0; i <= framecount; i++)
749  {
750  ProcessFrame(uimfReader, file, i);
751  }
752  }
753  else
754  {
755  ProcessFrame(uimfReader, file, options.Frame);
756  }
757  }
758 
759  Console.WriteLine("Finished processing " + file.FullName);
760  }
