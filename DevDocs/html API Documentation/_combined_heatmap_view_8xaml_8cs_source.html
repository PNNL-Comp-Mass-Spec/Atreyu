182  private async void CombinedHeatmapViewPreviewDrop(object sender, DragEventArgs e)
183  {
184  if (!e.Data.GetDataPresent(DataFormats.FileDrop, true))
185  {
186  return;
187  }
188 
189  var droppedFilePaths = e.Data.GetData(DataFormats.FileDrop, true) as string[];
190  if (droppedFilePaths == null || droppedFilePaths.Length < 1)
191  {
192  return;
193  }
194 
195  var file = droppedFilePaths[0];
196  if (!File.Exists(file))
197  {
198  return;
199  }
200 
201  await this.ViewModel.InitializeUimfData(file);
202  }

228  private async void MainTabControlPreviewDragEnter(object sender, DragEventArgs e)
229  {
230  var isCorrect = true;
231  string[] filenames = { };
232  if (e.Data.GetDataPresent(DataFormats.FileDrop, true))
233  {
234  filenames = (string[])e.Data.GetData(DataFormats.FileDrop, true);
235  foreach (string filename in filenames)
236  {
237  if (File.Exists(filename) == false)
238  {
239  isCorrect = false;
240  break;
241  }
242 
243  var info = new FileInfo(filename);
244 
245  if (info.Extension.ToLower() != ".uimf")
246  {
247  isCorrect = false;
248  break;
249  }
250  }
251  }
252 
253  e.Effects = isCorrect ? DragDropEffects.All : DragDropEffects.None;
254 
255  if (isCorrect)
256  {
257  await this.LoadFile(filenames[0]);
258  }
259 
260  e.Handled = true;
261  }
