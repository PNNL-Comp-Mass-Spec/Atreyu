162  private async void CombinedHeatmapViewPreviewDrop(object sender, DragEventArgs e)
163  {
164  if (!e.Data.GetDataPresent(DataFormats.FileDrop, true))
165  {
166  return;
167  }
168 
169  var droppedFilePaths = e.Data.GetData(DataFormats.FileDrop, true) as string[];
170  if (droppedFilePaths == null || droppedFilePaths.Length < 1)
171  {
172  return;
173  }
174 
175  var file = droppedFilePaths[0];
176  if (!File.Exists(file))
177  {
178  return;
179  }
180 
181  await this.ViewModel.InitializeUimfData(file);
182  }

208  private async void MainTabControlPreviewDragEnter(object sender, DragEventArgs e)
209  {
210  var isCorrect = true;
211  string[] filenames = { };
212  if (e.Data.GetDataPresent(DataFormats.FileDrop, true))
213  {
214  filenames = (string[])e.Data.GetData(DataFormats.FileDrop, true);
215  foreach (string filename in filenames)
216  {
217  if (File.Exists(filename) == false)
218  {
219  isCorrect = false;
220  break;
221  }
222 
223  var info = new FileInfo(filename);
224 
225  if (info.Extension.ToLower() != ".uimf")
226  {
227  isCorrect = false;
228  break;
229  }
230  }
231  }
232 
233  e.Effects = isCorrect ? DragDropEffects.All : DragDropEffects.None;
234 
235  if (isCorrect)
236  {
237  await this.LoadFile(filenames[0]);
238  }
239 
240  e.Handled = true;
241  }
