124  private static ImageFormat GetImageFormat(string fileName)
125  {
126  var extension = Path.GetExtension(fileName);
127  if (string.IsNullOrEmpty(extension))
128  {
129  throw new ArgumentException(
130  string.Format("Unable to determine file extension for fileName: {0}", fileName));
131  }
132 
133  switch (extension.ToLower())
134  {
135  case @".bmp":
136  return ImageFormat.Bmp;
137 
138  case @".gif":
139  return ImageFormat.Gif;
140 
141  case @".ico":
142  return ImageFormat.Icon;
143 
144  case @".jpg":
145  case @".jpeg":
146  return ImageFormat.Jpeg;
147 
148  case @".png":
149  return ImageFormat.Png;
150 
151  case @".tif":
152  case @".tiff":
153  return ImageFormat.Tiff;
154 
155  case @".wmf":
156  return ImageFormat.Wmf;
157 
158  default:
159  throw new NotImplementedException();
160  }
161  }

213  private void SaveExportedHeatmapCompressedData()
214  {
215  var filename = this.GetDataFilename();
216 
217  if (string.IsNullOrWhiteSpace(filename))
218  {
219  return;
220  }
221 
222  var temp = this.CombinedHeatmapViewModel.ExportHeatmapDataCompressed();
223 
224  using (var outfile = new StreamWriter(filename))
225  {
226  for (var x = 0; x < temp.GetLength(0); x++)
227  {
228  var content = string.Empty;
229  for (var y = 0; y < temp.GetLength(1); y++)
230  {
231  content += temp[x, y].ToString("0.00") + ",";
232  }
233 
234  outfile.WriteLine(content);
235  }
236  }
237  }
