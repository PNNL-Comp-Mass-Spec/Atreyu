486  public Image GetImage()
487  {
488  var tic = this.TotalIonChromatogramViewModel.GetTicImage();
489  var mz = this.MzSpectraViewModel.GetMzImage();
490  var heatmap = this.HeatMapViewModel.GetHeatmapImage();
491  const int Alignment = 25;
492

533  public async void SumFrames(FrameRange sumFrames)
534  {
535  if (sumFrames == null)
536  {
537  return;
538  }
539 
540  if (this.UimfData == null)
541  {
542  return;
543  }
544 
545  this.currentStartFrame = sumFrames.StartFrame < 1 ? 1 : sumFrames.StartFrame;
546 
547  this.currentEndFrame = sumFrames.EndFrame < 1 ? 1 : sumFrames.EndFrame;
548 
549  await

551  this.UimfData.CurrentMinBin,
552  this.UimfData.CurrentMaxBin,
553  this.currentStartFrame,
554  this.currentEndFrame,
