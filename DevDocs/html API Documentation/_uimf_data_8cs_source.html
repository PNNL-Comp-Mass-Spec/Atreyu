649  {

665 
666  this.checking = false;

690  public BinRange GetBinRangeForMzWindow(double centerMz, double partsPerMillionTolerance)
691  {
692  this.MzCenter = centerMz;
693  this.PartsPerMillion = partsPerMillionTolerance;
694  var range = new BinRange();
695 
696  if (this.dataReader == null || this.Calibrator == null)
697  {
698  range.StartBin = 0;
699  range.EndBin = this.CurrentMaxBin;
700  this.mzWindow = range;
701  return range;
702  }
703 
704  var mzOffset = this.MzCenter * (this.PartsPerMillion / 1000000.0);
705 
706  range.StartBin =
707  (int)(this.Calibrator.MZtoTOF(this.MzCenter - mzOffset) / this.dataReader.TenthsOfNanoSecondsPerBin);
708 
709  range.EndBin =
710  (int)(this.Calibrator.MZtoTOF(this.MzCenter + mzOffset) / this.dataReader.TenthsOfNanoSecondsPerBin);
711 
712  this.mzWindow = range;
713  return range;
714  }

739  public async Task<double[,]> ReadData(bool returnGatedData = false)
740  {
741  if (this.CurrentMaxBin < 1)
742  {
743  return new double[0, 0];
744  }
745 
746  if (this.endScan < 1)
747  {
748  return new double[0, 0];
749  }
750 
751  this.LoadingData = true;
752 
753  var frameParams = this.dataReader.GetFrameParams(this.startFrameNumber);
754 
755  if (frameParams == null)
756  {
757  // Frame number is out of range
758  this.FrameData = new double[0, 0];
759  }
760  else
761  {
762  this.TotalBins = this.CurrentMaxBin - this.CurrentMinBin + 1;
763 
764  this.ValuesPerPixelY = (int)(this.TotalBins / (double)this.mostRecentHeight);
765 
766  var totalScans = this.EndScan - this.StartScan + 1;
767  this.ValuesPerPixelX = (int)(totalScans / (double)this.mostRecentWidth);
768 
769  if (this.ValuesPerPixelY < 1)
770  {
771  this.ValuesPerPixelY = 1;
772  }
773 
774  if (this.ValuesPerPixelX < 1)
775  {
776  this.ValuesPerPixelX = 1;
777  }
778 
779  this.FrameSlope = frameParams.GetValueDouble(FrameParamKeyType.CalibrationSlope);
780  this.FrameIntercept = frameParams.GetValueDouble(FrameParamKeyType.CalibrationIntercept);
781 
782  this.FrameType = frameParams.GetValue(FrameParamKeyType.FrameType);
783  this.FrameIntercept = frameParams.GetValueDouble(FrameParamKeyType.CalibrationIntercept);
784 
785  await Task.Run(
786  () =>
787  {
788  this.Calibrator = this.dataReader.GetMzCalibrator(frameParams);
789 
790  var frametype = GetFrameType(this.frameType);
791  double[] mzs;
792  int[] intensities;
793  this.dataReader.GetSpectrum(
794  this.StartFrameNumber,
795  this.EndFrameNumber,
796  frametype,
797  this.StartScan,
798  this.EndScan,
799  out mzs,
800  out intensities);
801 
802  var temp = this.dataReader.AccumulateFrameData(
803  this.startFrameNumber,
804  this.EndFrameNumber,
805  false,
806  this.StartScan,
807  this.EndScan,
808  this.CurrentMinBin,
809  this.CurrentMaxBin,
810  (int)this.ValuesPerPixelX,
811  (int)this.ValuesPerPixelY);
812 
813  var arrayLength =
814  (int)Math.Round((this.CurrentMaxBin - this.currentMinBin + 1) / this.ValuesPerPixelY);
815 
816  var tof = new double[arrayLength];
817  var mz = new double[arrayLength];
818 
819  for (var i = 0; i < arrayLength; i++)
820  {
821  tof[i] = this.dataReader.GetPixelMZ(i);
822  mz[i] = this.calibrator.TOFtoMZ(tof[i] * 10);
823  }
824 
825  this.MzArray = mzs;
826 
827  this.MzIntensities = intensities;
828 
829  this.BinToMzMap = mz;
830 
831  this.FrameData = temp;
832  });
833  }
834 
835  this.GateData();
836 
837  this.LoadingData = false;
838  return returnGatedData ? this.GatedFrameData : this.FrameData;
839  }

1033  private void ProcessData(Range range)
1034  {
1035  switch (range.RangeType)
1036  {
1037  case RangeType.BinRange:
1038  var binRange = range as BinRange;
1039  if (binRange == null)
1040  {
1041  throw new ArgumentException(
1042  "Range has it's RangeType set to BinRange but cannot be cast to BinRange",
1043  "range");
1044  }
1045 
1046  int min, max;
1047 
1048  if (this.WindowMz && this.mzWindow != null)
1049  {
1050  min = this.mzWindow.StartBin;
1051  max = this.mzWindow.EndBin;
1052  }
1053  else
1054  {
1055  min = 0;
1056  max = this.MaxBins;
1057  }
1058 
1059  if (binRange.StartBin < min)
1060  {
1061  binRange.StartBin = min;
1062  }
1063 
1064  if (binRange.EndBin > max)
1065  {
1066  binRange.EndBin = max;
1067  }
1068 
1069  this.CurrentMinBin = binRange.StartBin;
1070  this.CurrentMaxBin = binRange.EndBin;
1071 
1072  break;
1073  case RangeType.FrameRange:
1074  var frameRange = range as FrameRange;
1075  if (frameRange == null)
1076  {
1077  throw new ArgumentException(
1078  "Range has it's RangeType set to FrameRange but cannot be cast to FrameRange",
1079  "range");
1080  }
1081 
1082  this.StartFrameNumber = frameRange.StartFrame;
1083  this.EndFrameNumber = frameRange.EndFrame;
1084  break;
1085  case RangeType.ScanRange:
1086  var scanRange = range as ScanRange;
1087  if (scanRange == null)
1088  {
1089  throw new ArgumentException(
1090  "Range has it's RangeType set to ScanRange but cannot be cast to ScanRange",
1091  "range");
1092  }
1093 
1094  this.StartScan = scanRange.StartScan;
1095  this.EndScan = scanRange.EndScan;
1096  break;
1097  default:
1098  throw new NotImplementedException(
1099  "Currently ProcessRangeData only supports types of BinRange, FrameRange, and ScanRange, "
1100  + "but you passed something else and it scared us too much to continue.");
1101  }
1102  }
