587  {
588  if (this.checking)
589  {
590  return;

592 
593  this.checking = true;
594  Range currentRange;
595 
596  while (this.RangeUpdateList.TryDequeue(out currentRange))

603

625  public async Task<double[,]> ReadData(bool returnGatedData = false)
626  {
627  if (this.CurrentMaxBin < 1)
628  {
629  return new double[0, 0];
630  }
631 
632  if (this.endScan < 1)
633  {
634  return new double[0, 0];
635  }
636 
637  var frameParams = this.dataReader.GetFrameParams(this.startFrameNumber);
638 
639  if (frameParams == null)
640  {
641  // Frame number is out of range
642  this.FrameData = new double[0, 0];
643  }
644  else
645  {
646  this.TotalBins = this.CurrentMaxBin - this.CurrentMinBin + 1;
647 
648  this.ValuesPerPixelY = (int)(this.TotalBins / (double)this.mostRecentHeight);
649 
650  var totalScans = this.EndScan - this.StartScan + 1;
651  this.ValuesPerPixelX = (int)(totalScans / (double)this.mostRecentWidth);
652 
653  if (this.ValuesPerPixelY < 1)
654  {
655  this.ValuesPerPixelY = 1;
656  }
657 
658  if (this.ValuesPerPixelX < 1)
659  {
660  this.ValuesPerPixelX = 1;
661  }
662 
663  this.FrameSlope = frameParams.GetValueDouble(FrameParamKeyType.CalibrationSlope);
664  this.FrameIntercept = frameParams.GetValueDouble(FrameParamKeyType.CalibrationIntercept);
665 
666  this.FrameType = frameParams.GetValue(FrameParamKeyType.FrameType);
667  this.FrameIntercept = frameParams.GetValueDouble(FrameParamKeyType.CalibrationIntercept);
668 
669  await Task.Run(
670  () =>
671  {
672  var frametype = this.GetFrameType(this.frameType);
673  double[] mzs;
674  int[] intensities;
675  this.dataReader.GetSpectrum(
676  this.StartFrameNumber,
677  this.EndFrameNumber,
678  frametype,
679  this.StartScan,
680  this.EndScan,
681  out mzs,
682  out intensities);
683 
684  var temp = this.dataReader.AccumulateFrameData(
685  this.startFrameNumber,
686  this.EndFrameNumber,
687  false,
688  this.StartScan,
689  this.EndScan,
690  this.CurrentMinBin,
691  this.CurrentMaxBin,
692  (int)this.ValuesPerPixelX,
693  (int)this.ValuesPerPixelY);
694 
695  var arrayLength =
696  (int)Math.Round((this.CurrentMaxBin - this.currentMinBin + 1) / this.ValuesPerPixelY);
697 
698  var tof = new double[arrayLength];
699  var mz = new double[arrayLength];
700  this.Calibrator = this.dataReader.GetMzCalibrator(frameParams);
701 
702  for (var i = 0; i < arrayLength; i++)
703  {
704  tof[i] = this.dataReader.GetPixelMZ(i);
705  mz[i] = this.calibrator.TOFtoMZ(tof[i] * 10);
706  }
707 
708  this.MzArray = mzs;
709 
710  this.MzIntensities = intensities;
711 
712  this.BinToMzMap = mz;
713 
714  this.FrameData = temp;
715  });
716  }
717 
718  this.GateData();
719 
720  return returnGatedData ? this.GatedFrameData : this.FrameData;
721  }

848  private void GateData()
849  {
850  if (this.LowGate <= 0)
851  {
852  this.GatedFrameData = this.FrameData;
853  return;
854  }
855 
856  var temp = new double[this.FrameData.GetLength(0), this.FrameData.GetLength(1)];
857 
858  for (var x = 0; x < temp.GetLength(0); x++)
859  {
860  for (var y = 0; y < temp.GetLength(1); y++)
861  {
862  if (this.FrameData[x, y] > this.LowGate && this.FrameData[x, y] < this.HighGate)
