320  public void UpdateFrameData(double[,] framedata)
321  {
322  if (this.uimfData == null)
323  {
324  return;
325  }
326 
327  if (framedata == null)
328  {
329  return;
330  }
331 
332  if (this.BinToMzMap == null)
333  {
334  return;
335  }
336 
337  this.frameData = framedata;
338  this.frameDictionary = new Dictionary<double, double>();
339  this.mzFrameData = new Dictionary<double, double>();
340 
341  for (var j = 0; j < this.frameData.GetLength(1); j++)
342  {
343  double index = j + this.startMzBin;
344  var mzIndex = this.BinToMzMap[j];
345 
346  for (var i = 0; i < this.frameData.GetLength(0); i++)
347  {
348  if (this.frameDictionary.ContainsKey(index))
349  {
350  this.frameDictionary[index] += this.frameData[i, j];
351  }
352  else
353  {
354  this.frameDictionary.Add(index, this.frameData[i, j]);
355  }
356 
357  if (this.mzFrameData.ContainsKey(mzIndex))
358  {
359  this.mzFrameData[mzIndex] += this.frameData[i, j];
360  }
361  else
362  {
363  this.mzFrameData.Add(mzIndex, this.frameData[i, j]);
364  }
365  }
366  }
367 
368  var series = this.MzPlotModel.Series[0] as LineSeries;
369 
370  if (series != null)
371  {
372  if (series.YAxis != null)
373  {
374  series.YAxis.Title = this.ShowMz ? "m/z" : "Bin";
375  }
376 
377  series.Points.Clear();
378  if (this.ShowMz)
379  {
380  foreach (var d in this.mzFrameData)
381  {
382  series.Points.Add(new DataPoint(d.Value, d.Key));
383  }
384  }
385  else
386  {
387  foreach (var d in this.frameDictionary)
388  {
389  series.Points.Add(new DataPoint(d.Value, d.Key));
390  }
391  }
392  }
393 
394  this.FindPeaks();
395 
396  this.MzPlotModel.InvalidatePlot(true);
397  }

423  private void FindPeaks()
424  {
425  this.mzPlotModel.Annotations.Clear();
426 
427  if (!this.ShowMz)
428  {
429  // This currently does not support peakfinding in bin mode due to potential headaches,
430  // But as it seems users never use that mode, it is a non-issue for now.
431  return;
432  }
433 
434  // Create a new dictionary to work with the peak finder
435  var tempFrameList = new List<KeyValuePair<double, double>>(this.uimfData.MaxBins);
436 
437  for (var i = 0; i < this.MzArray.Length && i < this.MzIntensities.Length; i++)
438  {
439  tempFrameList.Add(new KeyValuePair<double, double>(this.MzArray[i], this.MzIntensities[i]));
440  }
441 
442  var tempList = Utilities.PeakFinder.FindPeaks(tempFrameList, 3);
443 
444  // Put the points on the graph
445  foreach (var resolutionDatapoint in tempList.Peaks)
446  {
447  var resolutionString = resolutionDatapoint.ResolvingPower.ToString("F1", CultureInfo.InvariantCulture);
448 
449  var peakPoint = new OxyPlot.Annotations.PointAnnotation
450  {
451  Text = "R=" + resolutionString,
452  X = resolutionDatapoint.Intensity / 1.03125,
453  Y = resolutionDatapoint.PeakCenter,
454  ToolTip = resolutionDatapoint.ToString()
455  };
456  this.mzPlotModel.Annotations.Add(peakPoint);
457  }
458  }
