302  public void UpdateFrameData(double[,] framedata)
303  {
304  if (this.uimfData == null)
305  {
306  return;
307  }
308 
309  if (framedata == null)
310  {
311  return;
312  }
313 
314  if (this.BinToMzMap == null)
315  {
316  return;
317  }
318 
319  this.frameData = framedata;
320  this.frameDictionary = new Dictionary<double, double>();
321  this.mzFrameData = new Dictionary<double, double>();
322 
323  for (var j = 0; j < this.frameData.GetLength(1); j++)
324  {
325  double index = j + this.startMzBin;
326  var mzIndex = this.BinToMzMap[j];
327 
328  for (var i = 0; i < this.frameData.GetLength(0); i++)
329  {
330  if (this.frameDictionary.ContainsKey(index))
331  {
332  this.frameDictionary[index] += this.frameData[i, j];
333  }
334  else
335  {
336  this.frameDictionary.Add(index, this.frameData[i, j]);
337  }
338 
339  if (this.mzFrameData.ContainsKey(mzIndex))
340  {
341  this.mzFrameData[mzIndex] += this.frameData[i, j];
342  }
343  else
344  {
345  this.mzFrameData.Add(mzIndex, this.frameData[i, j]);
346  }
347  }
348  }
349 
350  var series = this.MzPlotModel.Series[0] as LineSeries;
351 
352  if (series != null)
353  {
354  if (series.YAxis != null)
355  {
356  series.YAxis.Title = this.ShowMz ? "m/z" : "Bin";
357  }
358 
359  series.Points.Clear();
360  if (this.ShowMz)
361  {
362  foreach (var d in this.mzFrameData)
363  {
364  series.Points.Add(new DataPoint(d.Value, d.Key));
365  }
366  }
367  else
368  {
369  foreach (var d in this.frameDictionary)
370  {
371  series.Points.Add(new DataPoint(d.Value, d.Key));
372  }
373  }
374  }

378  this.MzPlotModel.InvalidatePlot(true);

405  private void FindPeaks()
406  {
407  var datapointList = new List<ResolutionDatapoint>();
408  const int Precision = 10000;
409  this.mzPlotModel.Annotations.Clear();
410 
411  if (!this.ShowMz)
412  {
413  // This currently does not support peakfinding in bin mode due to potential headaches,
414  // But as it seems users never use that mode, it is a non-issue for now.
415  return;
416  }
417 
418  var peakDetector = new PeakDetector();
419 
420  var finderOptions = PeakDetector.GetDefaultSICPeakFinderOptions();
421 
422  List<double> smoothedY;
423 
424  // Create a new dictionary so we don't modify the original one
425  var tempFrameList = new List<KeyValuePair<int, double>>(this.uimfData.MaxBins);
426 
427  // We have to give it integers, but we need the mz, so we will multiply the mz by the precision
428  // and later get the correct value back by dividing it out again
429  for (var i = 0; i < this.MzArray.Length && i < this.MzIntensities.Length; i++)
430  {
431  tempFrameList.Add(
432  new KeyValuePair<int, double>((int)(this.MzArray[i] * Precision), this.MzIntensities[i]));
433  }
434 
435  // I am not sure what this does and need to talk to Matt Monroe, but in the example exe file that came with the library
436  // they used half of the length of the list in their previous examples and this seems to work on teh zoomed out version
437  // but not when we zoom in, it seems like an offset problem.
438  var originalpeakLocation = tempFrameList.Count / 2;
439 
440  var allPeaks = peakDetector.FindPeaks(
441  finderOptions,
442  tempFrameList.OrderBy(x => x.Key).ToList(),
443  originalpeakLocation,
444  out smoothedY);
445 
446  var topThreePeaks = allPeaks.OrderByDescending(peak => smoothedY[peak.LocationIndex]).Take(3);
447 
448  foreach (var peak in topThreePeaks)
449  {
450  const double Tolerance = 0.01;
451  var centerPoint = tempFrameList.ElementAt(peak.LocationIndex);
452  var offsetMz = centerPoint.Key; // + firstpoint.Key;
453  var intensity = centerPoint.Value;
454  var smoothedPeakIntensity = smoothedY[peak.LocationIndex];
455 
456  var realMz = (double)offsetMz / Precision;
457  var halfmax = smoothedPeakIntensity / 2.0;
458 
459  var currPoint = new KeyValuePair<int, double>(0, 0);
460  var currPointIndex = 0;
461  double leftMidpoint = 0;
462  double rightMidPoint = 0;
463 
464  var leftSidePeaks = new List<KeyValuePair<int, double>>();
465  for (var l = peak.LeftEdge; l < peak.LocationIndex && l < tempFrameList.Count; l++)
466  {
467  leftSidePeaks.Add(tempFrameList[l]);
468  }
469 
470  var rightSidePeaks = new List<KeyValuePair<int, double>>();
471  for (var r = peak.LocationIndex; r < peak.RightEdge && r < tempFrameList.Count; r++)
472  {
473  rightSidePeaks.Add(tempFrameList[r]);
474  }
475 
476  // find the left side half max
477  foreach (var leftSidePeak in leftSidePeaks)
478  {
479  var prevPoint = currPoint;
480  currPoint = leftSidePeak;
481  var prevPointIndex = currPointIndex;
482 
483  currPointIndex = tempFrameList.BinarySearch(currPoint, new KvpCompare<double>());
484 
485  if (smoothedY[currPointIndex] < halfmax)
486  {
487  continue;
488  }
489 
490  if (Math.Abs(smoothedY[currPointIndex] - halfmax) < Tolerance)
491  {
492  leftMidpoint = currPoint.Key;
493  continue;
494  }
495 
497  double a1 = prevPoint.Key;
498  double a2 = currPoint.Key;
499  double c = halfmax;
500  double b1 = smoothedY[prevPointIndex];
501  double b2 = smoothedY[currPointIndex];
502 
503  leftMidpoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
504  break;
505  }
506 
507  // find the right side of the half max
508  foreach (var rightSidePeak in rightSidePeaks)
509  {
510  var prevPoint = currPoint;
511  currPoint = rightSidePeak;
512  var prevPointIndex = currPointIndex;
513  currPointIndex = tempFrameList.BinarySearch(currPoint, new KvpCompare<double>());
514 
515  if (smoothedY[currPointIndex] > halfmax || smoothedY[currPointIndex] < 0)
516  {
517  continue;
518  }
519 
520  if (Math.Abs(smoothedY[currPointIndex] - halfmax) < Tolerance)
521  {
522  rightMidPoint = currPoint.Key;
523  continue;
524  }
525 
527  double a1 = prevPoint.Key;
528  double a2 = currPoint.Key;
529  double c = halfmax;
530  double b1 = smoothedY[prevPointIndex];
531  double b2 = smoothedY[currPointIndex];
532 
533  rightMidPoint = a1 + ((a2 - a1) * ((c - b1) / (b2 - b1)));
534  break;
535  }
536 
537  var correctedRightMidPoint = rightMidPoint / Precision;
538  var correctedLeftMidPoint = leftMidpoint / Precision;
539 
540  var resolution = realMz / (correctedRightMidPoint - correctedLeftMidPoint);
541 
542  var temp = new ResolutionDatapoint
543  {
544  Intensity = intensity,
545  Mz = realMz,
546  Resolution = resolution,
547  SmoothedIntensity = smoothedPeakIntensity
548  };
549  datapointList.Add(temp);
550  }
551 
552  var tempList =
553  datapointList.Where(x => !double.IsInfinity(x.Resolution)).OrderByDescending(x => x.Intensity).Take(10);
554 
555  // Put the points on the graph
556  foreach (var resolutionDatapoint in tempList)
557  {
558  var resolutionString = resolutionDatapoint.Resolution.ToString("F1", CultureInfo.InvariantCulture);
559  var annotationText = "Peak Location:" + resolutionDatapoint.Mz + Environment.NewLine + "Intensity:"
560  + resolutionDatapoint.Intensity + Environment.NewLine + "Resolution:"
561  + resolutionString;
562  var peakPoint = new OxyPlot.Annotations.PointAnnotation
563  {
564  Text = "R=" + resolutionString,
565  X = resolutionDatapoint.Intensity / 1.03125,
566  Y = resolutionDatapoint.Mz,
567  ToolTip = annotationText
568  };
569  this.mzPlotModel.Annotations.Add(peakPoint);
570  }
571  }
